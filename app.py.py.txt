import streamlit as st
import pandas as pd
import plotly.express as px

st.set_page_config(page_title="Customer Segmentation Dashboard", layout="wide")

st.title("Customer Segmentation Dashboard")

# Load dataset
df = pd.read_csv("Customer_segmentation.csv")

st.subheader("Dataset Preview")
st.dataframe(df.head())

st.subheader("Dataset Information")
st.write("Rows:", df.shape[0])
st.write("Columns:", df.shape[1])

# Display charts only if columns exist
if "Gender" in df.columns:
    st.subheader("Gender Distribution")
    fig = px.pie(df, names="Gender")
    st.plotly_chart(fig, use_container_width=True)

if "Age" in df.columns:
    st.subheader("Age Distribution")
    fig = px.histogram(df, x="Age")
    st.plotly_chart(fig, use_container_width=True)

if "Annual Income (k$)" in df.columns:
    st.subheader("Annual Income Distribution")
    fig = px.histogram(df, x="Annual Income (k$)")
    st.plotly_chart(fig, use_container_width=True)

if "Spending Score (1-100)" in df.columns:
    st.subheader("Spending Score Distribution")
    fig = px.histogram(df, x="Spending Score (1-100)")
    st.plotly_chart(fig, use_container_width=True)

st.success("Customer Segmentation Dashboard Created Successfully!")
